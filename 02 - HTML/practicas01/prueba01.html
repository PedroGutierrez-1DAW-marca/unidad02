package katas;
import java.util.Date;

public class Olimpiadas_rusas {
	static String r="";
	static int i;
	static Date time;
	public static void main(String[] args) {
		time = new Date();
		olimpiadasRusas(5);
		olimpiadasRusas(10);
		olimpiadasRusas(11);
		olimpiadasRusas(206788);
		System.out.println("Tiempo: " + ((new Date()).getTime() - time.getTime()) + "ms");
	}
	public static void olimpiadasRusas(int n){
		boolean f = true;
		for(;f;i++){
			if(r.length() > n){
				f = false;
				i--;
			}else
				r += i;
		}
		System.out.println(r.charAt(n));
	}
}
